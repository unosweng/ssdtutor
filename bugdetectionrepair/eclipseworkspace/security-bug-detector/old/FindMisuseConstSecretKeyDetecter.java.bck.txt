   @Deprecated
   void analyzeCallGraphOneDepth(ClassInstanceCreation ciCre) {
      List<?> args = ciCre.arguments();
      SimpleName arg1 = (SimpleName) args.get(0); // # 1st argument # key # byte[]
      SimpleName arg2 = (SimpleName) args.get(1); // # 2nd argument # algorithm # String
      IBinding arg1Binding = arg1.resolveBinding();
      IBinding arg2Binding = arg2.resolveBinding();
      int posArg1 = 0, posArg2 = 0;

      for (CallGraphNode iGraphNode : callgraph) {
         MethodDeclaration iCalleeDec = findMethodDec(iGraphNode.getNode());
         // # Find arguments from the call.
         List<?> parameters = iCalleeDec.parameters();
         List<IBinding> listCalleeBinding = new ArrayList<IBinding>();

         for (int i = 0; i < parameters.size(); i++) {
            Object iObj = parameters.get(i);
            if (iObj instanceof SingleVariableDeclaration) {
               SingleVariableDeclaration svDec = (SingleVariableDeclaration) iObj;
               IBinding iBinding = svDec.getName().resolveBinding();
               if (iBinding.equals(arg1Binding)) {
                  posArg1 = i;
               } else if (iBinding.equals(arg2Binding)) {
                  posArg2 = i;
               }
               listCalleeBinding.add(iBinding);
            }
         }

         List<IMethod> callers = iGraphNode.getIMethodCaller();

         for (IMethod iCaller : callers) {
            // # Find the call location in the caller.
            MethodDeclaration iCallerDec = findMethodDec(iCaller);
            MethodInvocation iMethodCall = getMethodCall(iCallerDec, iCalleeDec);
            List<?> argsMethodCall = iMethodCall.arguments();
            for (Object iObj : argsMethodCall) {
               if (iObj instanceof SimpleName) {
                  SimpleName iSPName = (SimpleName) iObj;
                  IBinding iSPNameBinding = iSPName.resolveBinding();

                  for (IBinding jBinding : listCalleeBinding) {
                     if (jBinding.equals(iSPNameBinding)) {
                        System.out.println("[DBG] ");
                     }
                  }
               }
            }

            Object objectArg1 = iMethodCall.arguments().get(posArg1);
            Object objectArg2 = iMethodCall.arguments().get(posArg2);

            // # Check the pattern.
            print(iGraphNode, iCallerDec, iMethodCall);
            System.out.println("[DBG] \tCheck ARG: " + objectArg1 + ", " + objectArg2);
         }
         break;
      }
   }

   @Deprecated
   MethodDeclaration findMethodDec(IMethod javaMethodElem) {
      MethodDeclaration[] methods = typeDec.getMethods();
      String iCallerName = javaMethodElem.getElementName();

      for (MethodDeclaration iMethodDec : methods) {
         String identifier = iMethodDec.getName().getIdentifier();
         if (identifier.equals(iCallerName)) {
            return iMethodDec;
         }
      }
      return null;
   }

   @Deprecated
   MethodInvocation getMethodCall(MethodDeclaration callerDec, MethodDeclaration calleeDec) {
      IMethodBinding calleeBinding = calleeDec.resolveBinding();

      callerDec.accept(new ASTVisitor() {
         public boolean visit(MethodInvocation n) {
            IMethodBinding methodBinding = n.resolveMethodBinding();
            if (methodBinding != null && methodBinding.equals(calleeBinding)) {
               methodInvoc = n;
            }
            return true;
         }
      });
      return methodInvoc;
   }
